    <script>
        let addedEquipment = {};

        function showSection(sectionId) {
            document.querySelectorAll('section').forEach(section => {
                section.classList.remove('active');
            });
            document.getElementById(sectionId).classList.add('active');
        }

        function addEquipment(name, price) {
            if (addedEquipment[name]) {
                addedEquipment[name].quantity++;
            } else {
                addedEquipment[name] = { price: price, quantity: 1 };
            }
            updateAddedEquipment();
        }

        function updateAddedEquipment() {
            const addedSection = document.getElementById('added-equipment');
            const totalItems = document.getElementById('total-added-items');
            addedSection.innerHTML = '';

            let totalCount = 0;
            if (Object.keys(addedEquipment).length === 0) {
                addedSection.innerHTML = '<p>No equipment added yet.</p>';
                totalItems.innerText = 'Total Equipment Added: 0';
                return;
            }

            Object.keys(addedEquipment).forEach(name => {
                const item = addedEquipment[name];
                totalCount += item.quantity;

                const div = document.createElement('div');
                div.classList.add('added-item');
                div.innerHTML = `
                    <span>${name} - ₹${item.price} per week</span>
                    <span>
                        <button onclick="changeQuantity('${name}', -1)">-</button>
                        ${item.quantity}
                        <button onclick="changeQuantity('${name}', 1)">+</button>
                    </span>
                `;
                addedSection.appendChild(div);
            });

            totalItems.innerText = `Total Equipment Added: ${totalCount}`;
        }

        function changeQuantity(name, change) {
            if (addedEquipment[name]) {
                addedEquipment[name].quantity += change;
                if (addedEquipment[name].quantity <= 0) {
                    delete addedEquipment[name];
                }
                updateAddedEquipment();
            }
        }
    </script>
